CREATE TABLE Passenger (
    id INT PRIMARY KEY AUTO_INCREMENT,
    surname VARCHAR(50) NOT NULL,
    firstname VARCHAR(50) NOT NULL,
    middlename VARCHAR(50),
    passportnumber VARCHAR(20) UNIQUE NOT NULL
);


CREATE TABLE Flight (
    id INT PRIMARY KEY AUTO_INCREMENT,
    trainnumber VARCHAR(20) NOT NULL,
    station VARCHAR(50) NOT NULL,
    DepartureTime DATETIME NOT NULL,
    ArrivalTime DATETIME NOT NULL
);

CREATE TABLE TrainTicket (
    id INT PRIMARY KEY AUTO_INCREMENT,
    idpassenger INT NOT NULL,
    idflight INT NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    seatnumber VARCHAR(10) NOT NULL,
    FOREIGN KEY (idpassenger) REFERENCES Passenger(id),
    FOREIGN KEY (idflight) REFERENCES Flight(id)
);
